class Mickey{
  //プロパティを定義
  float x,y;  //X座標とY座標
  float w,h;  //width, height
  
  //コンストラクタ
  Mickey( float spos, float xpos, float ypos )
  {
    //プロパティセット
    this.x = xpos;
    this.y = ypos;
    this.w = spos;
    this.h = spos * 0.8;
    
    //svgシェイプを読み込む
    PShape ms = loadShape("mickey.svg");
    
    shapeMode(CENTER);
    shape( ms, this.x, this.y, this.w, this.h );
  }
}
