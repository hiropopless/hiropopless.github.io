```js
function getElementsByClass(className) {
    className = ' ' + className + ' ';
    var all = document.getElementsByTagName('*');
    var elements = new Array();
    for (var i = 0; i < all.length; i++) {
        if (0 <= (' ' + all[i].className + ' ').indexOf(className)) {
            elements[elements.length] = all[i];
        }
    }
    return elements;
}
```
